```c
#include <iostream>

int main() {
  int T; int H;int M;int S;int i;
  scanf("%d %d %d",&H,&M,&S);
  scanf("%d", &T);

  i = H * 3600 + M * 60 + S - T;  // 초로 시간 전환
  H = i / 3600; // 3600초 = 1시간
  M = i % 3600 / 60;
  S = i % 60 % 60;

  
  printf("%d %d %d",H,M,S);
}
```
