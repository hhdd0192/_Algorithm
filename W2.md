n개의 정수를 입력 받고, 그 정수의 평균값에 따라서 다른 결과를 출력

```c
#include <iostream>

int main() {
  int n,sum;
  scanf("%d",&n);
  for(int i = 0; i < n;i++)
    {
      int temp;
      scanf("%d",&temp); //띄어쓰기로 나눠서 입력 
      sum += temp;
      
    }
  sum = sum / n;
  
  if(sum >= 80) {
    printf("easy");
    }
  else if(sum >= 60) {
    printf("normal");
    }
  else {
    printf("hard");
    }

}

```

두개의 배열 사이에 같은 자리에 같은 값인 것이 몇 개 있는지 출력
```c
#include <iostream>

int main() {
  int n,a[1000000],b[1000000];
  int count =0;
  scanf("%d",&n);
  for(int i = 0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
  for(int i = 0;i<n;i++)
    {
      scanf("%d",&b[i]);
      if(a[i] == b[i])
      {
        count ++;
      }
    }

  printf("%d",count);
}
```

      n /= 10;
      l++;

        }
      p = count;

      temp++;
